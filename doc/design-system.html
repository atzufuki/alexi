Usage
<!-- Basic usage -->

<!-- With size class -->
<svg class="alexi-icon alexi-icon-lg"><use href="#icon-rocket"/></svg>

<!-- In a button -->
<button class="alexi-btn alexi-btn-primary">
  <svg class="alexi-icon"><use href="#icon-plus"/></svg> Add Item
</button>
